\section{Vertical rules}
\begin{tabular}{|l|ll}
    \toprule
    Animal  & Food  & Size      \\
    \midrule
    dog     & meat  & medium    \\
    \cmidrule{1-1}
    horse   & hay   & large     \\
    frog    & flies & small     \\
    \bottomrule
\end{tabular}
